Examples

This is an example of how to compare 2 version numbers. 

//===================
//==  Versionning  ==
//===================
#include "Versionning/Version.h"

//===========
//==  STD  ==
//===========
#include <iostream>

int main(int /*argc*/, char** /*argv*/)
{
    Vers::Version v1(1, 3, 5, 8);
    Vers::Version v2(1, 3, 7, 2);

    if (v1 < v2)
        std::cout << "Version v" << v1 << " is less than version v" << v2 << std::endl;

    return 0;
}

Output : 

Version v1.3.5.8 is less than version v1.3.7.2
